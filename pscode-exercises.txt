define function three(input_1, input_2, input_3, input_4, input_5):
	set average = (input_1 + input_2 + input_3 + input_4 + input_5) divided by 5
	set maximum = 0
	set minimum = 0
	for each number in (input_1, input_2, input_3, input_4, input_5):
		if number > maximum:
			set maximum = number
		else if number < minimum:
			set minimum = number
	
	print "The average of these numbers is:" average
	print "The maximum is:" maximum
	print "The minimum is:" minimum
